function listingNameSort() {
    productList.sort((a, b) => {
        if (a.listingName < b.listingName)
            return -1;
        if (a.listingName > b.listingName)
            return 1;
        return 0;
    });

    renderProductTable();
}

function statusSort() {
    productList.sort((a, b) => {
        if (a.status < b.status)
            return -1;
        if (a.status > b.status)
            return 1;
        return 0;
    });

    renderProductTable();
}

// delete window =========================================
function deleteProduct(index) {
    currentTargetProduct = index;

    $('#deleteWin').modal('toggle');
}

function deleteProductFromService() {
    $('#deleteWin').modal('hide');

    if (currentTargetProduct > -1) {
        productList.splice(currentTargetProduct, 1);
        currentTargetProduct = -1;
        renderProductTable();

        notifUserWithAlert('#successNotif');
    } else {
        notifUserWithAlert('#errorNotif');
    }
}

function deployProduct(index) {
    currentTargetProduct = index;

    $('#deployWin').modal('toggle');

    deployFile = null;
    document.getElementById("fileUpload").value = "";
    $('#deployFileName').text('no file');
    $('#fileContent').text('no content');
    $('#fileVersion').text('no version');
    $('#deployURL').val('');
    $('#deployWinSubmit').prop('disabled', true);
}

function deployFileUpload(event) {

    if (event !== null) {
        deployFile = event.target.files[0];

        readDeployFileContent(deployFile, (success) => {
            console.log(success)
            checkDeploy();
        });
    }
}
